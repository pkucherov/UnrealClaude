class IProcessLauncher

    /** Check if an application with the given PID is running */

};

**OpenCodeHttpClient.h â€” IHttpClient interface ONLY (per D-22, D-27):**
Create file with copyright header. This file declares the IHttpClient interface. The production FOpenCodeHttpClient implementation will be added in Plan 03:
```cpp
#pragma once
#include "CoreMinimal.h"
#include "HttpModule.h"
#include "Interfaces/IHttpRequest.h"

class IHttpClient
{
public:
    virtual ~IHttpClient() = default;

    /** Synchronous-style GET. OutBody receives response body. Returns HTTP status code (-1 on connection failure). */
    virtual int32 Get(const FString& Url, FString& OutBody) = 0;

    /** Synchronous-style POST. OutResponse receives response body. Returns HTTP status code. */
    virtual int32 Post(const FString& Url, const FString& Body, FString& OutResponse) = 0;

    /** Begin SSE stream connection. StreamDelegate fires per-chunk on HTTP thread. OnDisconnected fires when connection closes. */
    virtual void BeginSSEStream(const FString& Url,
        TFunction<bool(void* Ptr, int64 Length)> StreamCallback,
        TFunction<void()> OnDisconnected) = 0;

    /** Abort the active SSE stream */
    virtual void AbortSSEStream() = 0;
};
```

Note: The `BeginSSEStream` signature uses `TFunction` instead of UE delegates for easier lambda usage in tests and production. The `StreamCallback` returns `bool` (matching `FHttpRequestStreamDelegate` signature â€” return `false` to abort).

#include "OpenCode/IProcessLauncher.h"
#include "OpenCode/OpenCodeHttpClient.h"

/** Mock process launcher for testing FOpenCodeServerLifecycle without real process spawning */
class FMockProcessLauncher : public IProcessLauncher
{
public:
    bool bLaunchShouldSucceed = true;
    bool bProcessIsRunning = true;
    bool bApplicationIsRunning = true;
    uint32 MockPID = 12345;
    int32 LaunchCallCount = 0;
    int32 TerminateCallCount = 0;
    FString LastLaunchParams;

    virtual FProcHandle LaunchProcess(
        const TCHAR* URL, const TCHAR* Params,
        bool bLaunchDetached, bool bLaunchHidden, bool bLaunchReallyHidden,
        uint32* OutProcessID, int32 PriorityModifier,
        const TCHAR* WorkingDirectory, void* PipeWriteChild) override
    {
        ++LaunchCallCount;
        LastLaunchParams = Params;
        if (OutProcessID) *OutProcessID = MockPID;
        if (bLaunchShouldSucceed) return FProcHandle(); // Valid-ish handle
        return FProcHandle(); // UE FProcHandle default â€” check bLaunchShouldSucceed via IsValid override
    }

    virtual bool IsProcRunning(FProcHandle& Handle) override { return bProcessIsRunning; }
    virtual void TerminateProc(FProcHandle& Handle, bool bKillTree) override { ++TerminateCallCount; }
    virtual bool IsApplicationRunning(uint32 ProcessId) override { return bApplicationIsRunning; }
};

/** Mock HTTP client for testing without real network calls */
class FMockHttpClient : public IHttpClient
{
public:
    int32 MockStatusCode = 200;
    FString MockResponseBody = TEXT("{\"healthy\":true,\"version\":\"0.1.0\"}");
    int32 GetCallCount = 0;
    int32 PostCallCount = 0;
    FString LastGetUrl;
    FString LastPostUrl;
    FString LastPostBody;
    TFunction<bool(void*, int64)> LastStreamCallback;
    TFunction<void()> LastDisconnectCallback;
    bool bSSEStreamActive = false;

    virtual int32 Get(const FString& Url, FString& OutBody) override
    {
        ++GetCallCount;
        LastGetUrl = Url;
        OutBody = MockResponseBody;
        return MockStatusCode;
    }

    virtual int32 Post(const FString& Url, const FString& Body, FString& OutResponse) override
    {
        ++PostCallCount;
        LastPostUrl = Url;
        LastPostBody = Body;
        OutResponse = MockResponseBody;
        return MockStatusCode;
    }

    virtual void BeginSSEStream(const FString& Url,
        TFunction<bool(void* Ptr, int64 Length)> StreamCallback,
        TFunction<void()> OnDisconnected) override
    {
        LastGetUrl = Url;
        LastStreamCallback = StreamCallback;
        LastDisconnectCallback = OnDisconnected;
        bSSEStreamActive = true;
    }

    virtual void AbortSSEStream() override { bSSEStreamActive = false; }

    /** Test helper: simulate receiving SSE data */
    void SimulateSSEData(const FString& Data)

        if (LastStreamCallback)

            LastStreamCallback((void*)Utf8.Get(), Utf8.Length());
        }
    }

    /** Test helper: simulate SSE disconnect */
    void SimulateSSEDisconnect()
    {
        bSSEStreamActive = false;
        if (LastDisconnectCallback) LastDisconnectCallback();
    }
};
